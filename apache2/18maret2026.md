
# Install apache2 di ubuntu

 
```bash

sudo apt update
sudo apt-get install apache2 php php-xmlrpc php-mysql php-gd php-cli 
php-curl \
libphp-adodb libgd-dev php-pear \
php-common libapache2-mod-php php-fpm \
php-bz2 php-xml imagemagick git php-mbstring \
mariadb-server mariadb-client \
php8.3-intl

sudo a2enmod php8.3
sudo a2enmod proxy_fcgi setenvif
sudo a2enconf php8.3-fpm

sudo service apache2 stop
sudo service apache2 start
sudo systemctl reload apache2
systemctl status apache2

```
